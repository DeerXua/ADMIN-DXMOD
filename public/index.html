Code Lua tích hợp (thêm vào BRPlayerCharacterBase.lua)
-- ============ VPS UID CHECK (chèn sau dòng khai báo expireTime) ============
local VPS_API_URL = "http://YOUR_VPS_IP:3000/api/check"
local VPS_API_KEY = "DX_API_KEY_2026"
local _UID_CACHE = { valid = false, checked_at = 0, expire_ts = 0 }
local CACHE_TTL  = 60       -- giây
local OFFLINE_GRACE = 86400 -- 1 ngày fallback offline

local function _XorEncode(str, key)
    local out = {}
    for i = 1, #str do
        local k = key:byte(((i-1) % #key) + 1)
        out[i] = string.char(string.byte(str, i) ~ k)
    end
    return table.concat(out)
end

local function CheckUIDWithServer()
    local now = os.time()
    -- Trả về cache nếu còn hiệu lực
    if now - _UID_CACHE.checked_at < CACHE_TTL then
        return _UID_CACHE.valid
    end

    local uid = ""
    pcall(function()
        local GameplayData = package.loaded["GameLua.GameCore.Data.GameplayData"]
        local PC = GameplayData and GameplayData.GetPlayerController()
        if PC then
            local PS = PC:GetPlayerStateSafety()
            if slua.isValid(PS) then
                uid = tostring(PS.PlayerId or PS.UniqueId or "")
            end
        end
    end)

    if uid == "" then
        _UID_CACHE.valid = _UID_CACHE.expire_ts > now
        return _UID_CACHE.valid
    end

    local success, result = pcall(function()
        -- slua HTTP client (nếu khả dụng)
        if slua and slua.HttpClient then
            local body = '{"uid":"' .. uid .. '","apiKey":"' .. VPS_API_KEY .. '"}'
            local resp = slua.HttpClient:Post(VPS_API_URL, body, "application/json")
            return resp
        end
    end)

    if success and result then
        pcall(function()
            local active = string.find(result, '"active":true') ~= nil
            local expStr = string.match(result, '"expire_time":"([^"]+)"')
            _UID_CACHE.valid      = active
            _UID_CACHE.checked_at = now
            if expStr then
                -- parse ISO date cơ bản (năm-tháng-ngày)
                local y, mo, d = expStr:match("(%d+)-(%d+)-(%d+)")
                if y then
                    _UID_CACHE.expire_ts = os.time({ year=tonumber(y), month=tonumber(mo), day=tonumber(d), hour=23, min=59, sec=59 })
                end
            end
            _G.UID_VALID = active
        end)
    else
        -- Fallback offline: dùng expire cũ hoặc grace period
        _UID_CACHE.checked_at = now
        if _UID_CACHE.expire_ts > 0 then
            _UID_CACHE.valid = _UID_CACHE.expire_ts > now
        else
            -- Lần đầu không kết nối được: cho phép 1 ngày
            _UID_CACHE.valid      = true
            _UID_CACHE.expire_ts  = now + OFFLINE_GRACE
        end
        _G.UID_VALID = _UID_CACHE.valid
    end

    return _UID_CACHE.valid
end
